176 static inline void hw_otpc_init(void)
177 {
178  GLOBAL_INT_DISABLE();
179  if ((dg_configBLACK_ORCA_IC_REV == BLACK_ORCA_IC_REV_A)
180  || ((dg_configUSE_AUTO_CHIP_DETECTION == 1) && (CHIP_IS_AD || CHIP_IS_AE))) {
181  /*
182  * Reset OTP controller
183  */
184  OTPC->OTPC_MODE_REG = HW_OTPC_MODE_STBY; // Set OTPC to standby
185  REG_SET_BIT(CRG_TOP, SYS_CTRL_REG, OTPC_RESET_REQ); // Reset the OTPC
186  REG_CLR_BIT(CRG_TOP, SYS_CTRL_REG, OTPC_RESET_REQ); // Get the OTPC out of Reset
187 
188  /*
189  * Initialize the POR registers
190  */
191  OTPC->OTPC_NWORDS_REG = OTPC_NWORDS_REG_RESET;
192  OTPC->OTPC_TIM1_REG = OTPC_TIM1_REG_RESET;
193  OTPC->OTPC_TIM2_REG = OTPC_TIM2_REG_RESET;
194  }
195 
196  /*
197  * Enable OTPC clock
198  */
199  CRG_TOP->CLK_AMBA_REG |= 1 << REG_POS(CRG_TOP, CLK_AMBA_REG, OTP_ENABLE);
200  GLOBAL_INT_RESTORE();
201 }
